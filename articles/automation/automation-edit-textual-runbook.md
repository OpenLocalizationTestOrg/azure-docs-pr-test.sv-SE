<span data-ttu-id="56c2a-185">Följande exempelkommandon visar hur du hämtar skriptet för en runbook och spara den till en skriptfil.</span><span class="sxs-lookup"><span data-stu-id="56c2a-185">The following sample commands show how to retrieve the script for a runbook and save it to a script file.</span></span> <span data-ttu-id="56c2a-186">I det här exemplet hämtas utkastversionen.</span><span class="sxs-lookup"><span data-stu-id="56c2a-186">In this example, the Draft version is retrieved.</span></span> <span data-ttu-id="56c2a-187">Det är också möjligt att hämta den publicerade versionen av runbooken även om den här versionen inte kan ändras.</span><span class="sxs-lookup"><span data-stu-id="56c2a-187">It is also possible to retrieve the Published version of the runbook although this version cannot be changed.</span></span>

    $automationAccountName = "MyAutomationAccount"
    $runbookName = "Sample-TestRunbook"
    $scriptPath = "c:\runbooks\Sample-TestRunbook.ps1"

    $runbookDefinition = Get-AzureAutomationRunbookDefinition -AutomationAccountName $automationAccountName -Name $runbookName -Slot Draft
    $runbookContent = $runbookDefinition.Content

    Out-File -InputObject $runbookContent -FilePath $scriptPath
